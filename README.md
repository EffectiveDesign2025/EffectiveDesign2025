<script>
    // Ajusta a altura do iframe automaticamente
    function ajustarAltura() {
        const section = document.getElementById('servicos-section');
        const altura = section.scrollHeight;
        parent.postMessage({ 'height': altura }, '*');
    }

    // Observa mudanças no conteúdo
    const observer = new ResizeObserver(ajustarAltura);
    observer.observe(document.getElementById('servicos-section'));

    // Ajusta ao carregar
    window.addEventListener('load', ajustarAltura);
